go语言中不同的并发模型

## ping-pong模型，更准确的应该是按顺序传递模型
多个人围成一个圆将球依次进行传递
```
import (
	"time"
	"fmt"
	"github.com/petermattis/goid"
)

func main() {
	var Ball int
	table := make(chan int)
	for i := 0; i < 10; i++{
		go player(table)
	}
	table <- Ball
	time.Sleep(1 * time.Second)
	<-table
}

func player(table chan int) {
	for {
		ball := <-table
		ball++
		fmt.Printf("gid: %d, ball: %d\n",goid.Get(),ball)
		time.Sleep(100 * time.Millisecond)
		table <- ball
	}
}
```
